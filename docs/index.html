import { Client } from "@modelcontextprotocol/sdk/client";

// Call Ensample as a standard tool
const result = await client.callTool({
    name: "ensample.execute_llm_role",
    arguments: {
        prompt: "Generate critical system config...",
        voting_k: 3, // Require 3 votes agreement
        ensemble_config: {
            models: [
                { provider: "openai", model: "gpt-4o" },
                { provider: "anthropic", model: "claude-3-5" }
            ]
        }
    }
});

console.log(result.confidence_score); // 0.99
